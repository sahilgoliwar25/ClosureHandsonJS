function counter(){
    var counter = 0;

    function IncreaseCounter() {
        return counter += 1;
    };

    return IncreaseCounter;
}

var counter = counter();
alert(counter());
alert(counter());
alert(counter());
alert(counter());

In the above code, the outer function counter returns the reference of inner function "IncreaseCounter()". IncreaseCounter increases the outer variable counter to one and returns the counter. So everytime inner function is called, counter value will be incremented by one.

Q4. Write a code to calculate area of a rectangle using inner function. In this case outer function should accept parameter length and inner function should accept parameter breadth.

=>

function question4() {
  function areaRectangle(length) {
    function inner(breadth) {
      return length * breadth;
    }
    return inner;
  }
  const area = areaRectangle(10);
  const output = area(20);
  console.log(output);
}

In the above code, the outer function areaRectangle returns the reference of inner function "inner()". length is passed as a parameter in outer function whereas breadth is passed as a parameter in inner function. Inner function multiplies length and breadth and returns the output.

Q5. Take a variable in outer function and create an inner function to increase the counter every time it is called

=>

function Counter() {
    var counter = 0;

    function IncreaseCounter() {
      return (counter += 1);
    }

    return IncreaseCounter;
  }

  var counter = Counter();
  alert(counter()); // 1
  alert(counter()); // 2
  alert(counter()); // 3
  alert(counter()); // 4
  alert(counter()); // 5

In the above code, the outer function counter returns the reference of inner function "IncreaseCounter()". Variable counter is declared in the outer function. IncreaseCounter increases the outer variable counter to one and returns the counter. So everytime inner function is called, counter value will be incremented by one.
